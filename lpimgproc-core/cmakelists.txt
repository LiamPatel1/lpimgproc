add_library(lpimgproc-core STATIC "src/Image.cpp" "include/lpimgproc/Operators.h" "src/operators.cpp" "include/lpimgproc/Kernel.h")

target_include_directories(lpimgproc-core PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/include)

target_include_directories(lpimgproc-core
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include>
)

install(TARGETS lpimgproc-core
    EXPORT lpimgprocTargets
    ARCHIVE DESTINATION lib
    LIBRARY DESTINATION lib
    RUNTIME DESTINATION bin
    INCLUDES DESTINATION include
)
install(DIRECTORY include/ DESTINATION include)
